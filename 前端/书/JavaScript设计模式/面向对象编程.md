## 面向对象编程

```js
const Book = (function() {
  //静态私有变量
  let bookNum = 0;
  //静态私有方法
  function() checkBook(name) {}
  //创建类
  function _book(newId, newName, newPrice) {
    //私有变量
    let name, price;
    //私有方法
    function checkId() {};
    //特权方法
    this.getName = function() {};
    this.setName = function() {};
    //公有属性
    this.id = newId;
    //公有方法
    this.copy = function() {};
    //构造器
    this.setName(name);
  }
  _book.prototype = {
    //静态公有属性
    isJSBook : false;
    //静态公有方法
    display : function() {}
  }
  //返回类
  return _book
})

//类静态公有属性（对象不能访问）
Book.isChisese = true
//类静态公有方法（对象不能访问）
Book.resetTime = function() {
  console.log('new Tiem')
}
```
